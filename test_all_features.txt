# 1. 新しい構文（型制約）をプログラムする
syntax positive $x = { $x > 0 }
syntax even $x = { $x / 2 * 2 == $x }

# 2. 関数（ラムダ式）の定義
# もし x が 10 より大きければ x+5、そうでなければ x*2 を返す
logic_func = \x -> if x > 10 then x + 5 else x * 2

# 3. 制約の適用
positive a
even a
a < 20

# 4. 関数の適用と逆算
# 結果が 20 になるような入力 'a' は何か？
# (もし a > 10 なら a+5=20 => a=15 (奇数なのでNG)
#  もし a <= 10 なら a*2=20 => a=10 (偶数・正・10以下なのでOK!)
res = logic_func(a)

# ここで結果を固定して、a を逆算させる
res == 20

# 5. 論理演算のテスト
# a は 10 であるか、あるいは 100 である
(a == 10) || (a == 100)
